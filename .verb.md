## Usage

```js
var last = require('array-last');

union(['a'], ['b', 'c'], ['d', 'e', 'f']);
//=> ['a', 'b', 'c', 'd', 'e', 'f']
```

Returns only unique elements:

```js
union(['a', 'a'], ['b', 'c']);
//=> ['a', 'b', 'c']
```
